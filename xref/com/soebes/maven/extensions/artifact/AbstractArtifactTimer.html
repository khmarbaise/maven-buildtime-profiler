1   package com.soebes.maven.extensions.artifact;

3   import java.util.Map;
4   import java.util.concurrent.ConcurrentHashMap;
5   
6   import org.eclipse.aether.RepositoryEvent;
7   import org.eclipse.aether.artifact.Artifact;
8   
9   import com.soebes.maven.extensions.TimePlusSize;
10  
11  /**
12   * @author Karl Heinz Marbaise <a href="mailto:khmarbaise@apache.org">khmarbaise@apache.org</a>
13   *
14   */
15  public abstract class AbstractArtifactTimer
16  {
17      private Map<String, TimePlusSize> timerEvents;
18  
19      public AbstractArtifactTimer()
20      {
21          this.timerEvents = new ConcurrentHashMap<>();
22      }
23      
24      protected String getArtifactId( Artifact artifact )
25      {
26          StringBuilder sb = new StringBuilder( 128 );
27          sb.append( artifact.getGroupId() );
28          sb.append( ":" ).append( artifact.getArtifactId() ).append( ":" ).append( artifact.getVersion() );
29          if ( artifact.getClassifier() != null )
30          {
31              sb.append( ':' ).append( artifact.getClassifier() );
32          }
33          sb.append( ':' ).append( artifact.getExtension() );
34          return sb.toString();
35      }
36  
37      protected Map<String, TimePlusSize> getTimerEvents()
38      {
39          return timerEvents;
40      }
41  
42      public void start( RepositoryEvent event )
43      {
44          String artifactId = getArtifactId( event.getArtifact() );
45          TimePlusSize systemTime = new TimePlusSize();
46          systemTime.start();
47          getTimerEvents().put( artifactId, systemTime );
48      }
49  
50      public void stop( RepositoryEvent event )
51      {
52          String artifactId = getArtifactId( event.getArtifact() );
53          if ( !getTimerEvents().containsKey( artifactId ) )
54          {
55              throw new IllegalArgumentException( "Unknown artifactId (" + artifactId + ")" );
56          }
57          getTimerEvents().get( artifactId ).stop();
58          getTimerEvents().get( artifactId ).setSize( event.getArtifact().getFile().length() );
59      }
60  
61      private final double MiB = 1024 * 1024;
62  
63      protected double calculateMegabytesPerSeconds( long timeInMilliseconds, long sizeInBytes )
64      {
65          double dividerTime = ( timeInMilliseconds / 1000.0 );
66          return (double) sizeInBytes / dividerTime / MiB;
67      }
68  
69  }
